CREATE TABLE CAR_SHOWROOM (
CAR_id INT PRIMARY KEY,
Price INT,
Colour VARCHAR(25),
Brand VARCHAR(25),
Type_body VARCHAR(25),
Volume_engine DOUBLE,
Transmission VARCHAR(25),
Release_date DATE
);

CREATE TABLE STAFF(
EMPLOYEE_id INT PRIMARY KEY,
Name_emp VARCHAR (50),
Surname VARCHAR (50),
Position_emp VARCHAR(250),
email VARCHAR (50),
Address VARCHAR(50),
Phone_number INT
);

CREATE TABLE COMPANY (
id_C INT PRIMARY KEY,
Name_comp VARCHAR(50),
Country VARCHAR,
Car_id INT REFERENCES CAR_SHOWROOM (CAR_id)
);

CREATE TABLE IN_STOCK (
Car_id INT REFERENCES CAR_SHOWROOM (CAR_id)
);

CREATE TABLE SERVICE (
Employee_id INT REFERENCES STAFF (EMPLOYEE_id),
Customers_id INT REFERENCES CUSTOMERS (id_c)
);

CREATE TABLE CUSTOMERS (
id_c INT PRIMARY KEY,
Name_cus VARCHAR(50),
Surname VARCHAR(50),
email VARCHAR(50),
City VARCHAR(50),
Address VARCHAR(50),
Phone_number INT
);

CREATE TABLE SALE (
Date_sale DATE,
Car_id INT REFERENCES CAR_SHOWROOM (CAR_id),
Customers_id INT REFERENCES CUSTOMERS (id_c),
Order_id INT REFERENCES Order_cust (id_O) 
);

CREATE TABLE Order_cust(
id_O INT PRIMARY KEY,
Car_id INT REFERENCES CAR_SHOWROOM (CAR_id),
Customers_id INT REFERENCES CUSTOMERS (id_c)
);
